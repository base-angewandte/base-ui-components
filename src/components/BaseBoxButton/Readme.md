const executeFire = function(){
    alert('button clicked!');
}
<div 
  style="background-color: rgb(240, 240, 240); padding: 16px; display:flex;">

       style="margin-right: 16px;"
       box-style="large"
       text="Large Style Button"
       icon="sheet-empty"
       subtext="Click or Drag'n Drop"

 <base-box-button
